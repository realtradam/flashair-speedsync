<script lang="ts">
  import { imageCache } from '../cache';

  interface Stats {
    memoryEntries: number;
    memoryFullCount: number;
    memoryThumbCount: number;
    idbEntries: number;
    idbFullCount: number;
    idbThumbCount: number;
    idbTotalBytes: number;
    idbError: string | undefined;
    storageEstimate: { usage: number; quota: number } | undefined;
  }

  let stats = $state<Stats | undefined>(undefined);
  let refreshing = $state(false);

  function formatBytes(bytes: number): string {
    if (bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    const val = bytes / Math.pow(k, i);
    return `${val.toFixed(1)} ${sizes[i]}`;
  }

  async function refresh() {
    refreshing = true;
    try {
      stats = await imageCache.getStats();
    } catch {
      // ignore
    } finally {
      refreshing = false;
    }
  }

  // Auto-refresh every 2 seconds
  $effect(() => {
    void refresh();
    const id = setInterval(() => {
      void refresh();
    }, 2000);
    return () => clearInterval(id);
  });
</script>
